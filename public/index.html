function calcYTM(price, flows, settlement) {
  const MS_YR = 365.25 * 24 * 3600 * 1000;
  let r = 0.10;
  for (let i = 0; i < 100; i++) {
    let pv = 0, dpv = 0;
    for (const f of flows) {
      const t = (f.fecha - settlement) / MS_YR;
      if (t <= 0) continue;
      const d = Math.pow(1 + r, t);
      pv += f.monto / d;
      dpv -= t * f.monto / (d * (1 + r));
    }
    if (Math.abs(pv - price) < 0.0001) break;
    r -= (pv - price) / dpv;
    r = Math.max(-0.5, Math.min(2, r));
  }
  return r * 100;
}
Formula teorica
// Newton-Raphson: encontrar r tal que
Precio / 100 = Σ [ Flujo(i) / (1 + r)t(i) ]

// Iteracion
PV(r) = Σ [ Flujo(i) / (1+r)t(i) ]
dPV(r) = Σ [ −t(i) × Flujo(i) / (1+r)t(i)+1 ]
r_nuevo = r − (PV − Precio) / dPV
// Converge cuando |PV − Precio| < 0.0001

Duration_Macaulay = Σ[ t(i) × VP(i) ] / Σ[ VP(i) ]
donde VP(i) = Flujo(i) / (1 + TIR)t(i)
# Python
from scipy.optimize import brentq
from datetime import date

def valor_presente(r, flujos, settlement):
  pv = 0
  for f in flujos:
    t = (f["fecha"] - settlement).days / 365.25
    if t > 0: pv += f["monto"] / (1 + r) ** t
  return pv

precio_norm = precio / 100
tir = brentq(lambda r: valor_presente(r, flujos, sett) - precio_norm, -0.5, 2.0)
tir_pct = tir * 100

Convergencia: 100 iteraciones max, threshold 0.0001, bounds [-50%, 200%].
Alternativa Python: scipy.optimize.brentq es mas robusto que Newton-Raphson manual.

Validar con negocio — calculos BOTE
① Confirmar que Precio/100 = Σ[flujo(i) / (1+r)^t(i)] — Newton-Raphson converge.
② Confirmar que Duration = Σ[t(i) × VP(i)] / Σ[VP(i)]. Si Duration = 1.5, suba de 1% en tasa → baja de ~1.5% en precio.
③ Confirmar que los flujos son monto fijo por 100 VN (cupon + amortizacion en el ultimo).

Nota dev
Orden de implementacion sugerido: LECAP → BOTE → BONCER → LETAM → BONTAM. Cada paso agrega una capa de complejidad.
Newton-Raphson reusable: la misma funcion calcYTM se usa en BOTE, BONCER, Soberanos y BONTAM. Solo cambia como se generan los flujos.
TAMAR acumulada por periodo: la simplificacion TAMAR_periodo ≈ TAMAR_TNA × base funciona para la estimacion, pero en produccion habria que acumular dia a dia dentro de cada periodo de cupon. 
const settlement = getSettlement(); // T+1
const tasaEfectiva = tamarTNA / 100 + spread;
let amortAcum = 0;
const flujos = cronograma.map(f => {
  const vr = 1 - amortAcum;
  const cupon = vr * tasaEfectiva * f.base;
  amortAcum += f.amortizacion;
  return { fecha: new Date(f.fecha), monto: cupon + f.amortizacion };
}).filter(f => f.fecha > settlement);
const tir = calcYTM(precio / 100, flujos, settlement);
